   ```typescript
   app.post("/hooks/post-tool-use", (req, res) => {
     const { tool_name, tool_input, session_id } = req.body;
     broadcast({
       type: "tool_event",
       toolName: tool_name || "unknown",
       input: tool_input || {},
       sessionId: session_id || "",
       timestamp: Date.now(),
     });
     res.json({});
   });
   ```
